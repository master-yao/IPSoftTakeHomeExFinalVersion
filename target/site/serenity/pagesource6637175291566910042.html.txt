<script>
/**
 * This script does not check the actual requirements on Guidelines page. It only
 * checks whether the user's global edit count is more than MIN_EDIT_COUNT and whether
 * the global account was registered before MIN_REGISTRATION. If the user's global account
 * registration time is newer, then we also try to find whether there is a local account
 * which was registered before the minimum registration time. Users who do not match these
 * criteria are considered as ineligible.
 */
var stewardVoteBanner = ( function( $, mw ) {
  var MIN_REGISTRATION = '2016-11-01T00:00:00Z',
    MIN_EDIT_COUNT = 600,
    COOKIE_KEY = 'StewardVoteEligible',
    username = mw.config.get( 'wgUserName' ),
    $notice = $( '#stewvote' );

  function checkEligibility() {
    getGlobalAccountInfo().done( function( globalInfo ) {
        var eligibility = isEligible( globalInfo );
        setDisplayAndCookie( eligibility );
     } ).fail( function() {
        // We don't have the required info so this defaults to the hidden banner
        mw.log( 'API request failed for steward vote CNbanner' );
    } );
  }

  function getGlobalAccountInfo() {
    return new mw.Api().get( {
        action: 'query',
        meta: 'globaluserinfo',
        guiprop: [ 'editcount', 'merged' ],
        guiuser: username
      } ).then( function( result ) {
        return result.query.globaluserinfo;
      }, function() {
        return null;
      } );
  }

  function isEligible( globalInfo ) {
    if ( globalInfo.editcount &lt; MIN_EDIT_COUNT ) {
      // User doesn't have the required number of edits globally
      return false;
    }
    if ( globalInfo.registration &lt;= MIN_REGISTRATION ) {
      // Global account was created before minimum registration
      // (and above global editcount threshold) so this user is eligible
      return true;
    }

    // Global account registration time may be newer than registration time for
    // a local account so we'll iterate through all local accounts and find
    // the oldest account to compare with minimum registration time
    var wikis = globalInfo.merged,
        wikiCount = wikis.length,
        oldestRegistration = wikis[0].registration; // default to first account on wikis array
    for ( var i = 0; i &lt; wikiCount; i++ ) {
        if ( wikis[i].registration &lt; oldestRegistration ) {
            oldestRegistration = wikis[i].registration;
        }
        if ( oldestRegistration &lt;= MIN_REGISTRATION ) {
            // A local account registered before the minimum registration
            // time so this user is eligible
            return true;
        }
    }

    // We couldn't find a local account registered before the minimum
    // registration time - ultimately the user is not eligible to vote
    return false;
  }

  function setDisplayAndCookie( eligible ) {
    $notice.css( 'display', eligible ? 'block' : 'none' );
    $.cookie(
      COOKIE_KEY,
      eligible ? 1 : 0,
      { expires: 29 }
    );
  }

  return {
    initialize: function() {
      // Not logged in but just to be safe - this shouldn't actually
      // happen because the banner should only be loaded for logged-in users
      if ( username === null ) {
        mw.log( 'steward vote CNbanner loaded on an anon user - not showing' );
        return;
      }
      if ( $notice.length === 0 ) {
        mw.log( 'Could not locate #stewvote' );
        return;
      }

      mw.loader.using( 'jquery.cookie', function() {
        var isEligible = $.cookie( COOKIE_KEY );
        if ( isEligible !== null ) {
          // Eligibility already checked so use the value from the cookie and be done
          $notice.css( 'display', ( isEligible === '1' ) ? 'block' : 'none' );
          return;
        }
        // Not checked yet so do the checks and store in the cookie and show/hide accordingly
        mw.loader.using( 'mediawiki.api', checkEligibility );
      } );
    }
  };
}( jQuery, mediaWiki ) );
stewardVoteBanner.initialize();

<h2 style="margin:3px; background:#cef2e0; font-family:inherit; font-size:120%; font-weight:bold; border:1px solid #a3bfb1; text-align:left; color:#000; padding:0.2em 0.4em;" id="mp-tfa-h2"><span id="From_today.27s_featured_article" class="mw-headline">From today's featured article</span></h2>

<td style="color:#000;">
<div style="padding:2px 5px" id="mp-tfa">
<div style="float: left; margin: 0.5em 0.9em 0.4em 0em;" id="mp-tfa-img">
<div style="background: transparent; border: none; padding: 0; max-width: 100px;" class="thumbinner mp-thumb"><a title="Doris Lessing in 2006" class="image" href="/wiki/File:Doris_Lessing_3.jpg"><img width="100" height="125" data-file-height="740" data-file-width="592" srcset="//upload.wikimedia.org/wikipedia/commons/thumb/2/26/Doris_Lessing_3.jpg/150px-Doris_Lessing_3.jpg 1.5x, //upload.wikimedia.org/wikipedia/commons/thumb/2/26/Doris_Lessing_3.jpg/200px-Doris_Lessing_3.jpg 2x" src="//upload.wikimedia.org/wikipedia/commons/thumb/2/26/Doris_Lessing_3.jpg/100px-Doris_Lessing_3.jpg" alt="Doris Lessing in 2006" /></a>
<div style="padding: 0.25em 0; word-wrap: break-word;" class="thumbcaption">Doris Lessing</div>
</div>
</div>
<p><i><b><a title="The Good Terrorist" href="/wiki/The_Good_Terrorist">The Good Terrorist</a></b></i> is a 1985 <a title="Political fiction" href="/wiki/Political_fiction">political novel</a> by <a title="Doris Lessing" href="/wiki/Doris_Lessing">Doris Lessing</a> <i>(pictured)</i>, a winner of the <a title="Nobel Prize in Literature" href="/wiki/Nobel_Prize_in_Literature">Nobel Prize in Literature</a>. The story examines events in the life of Alice, a naïve and well-intentioned <a title="Squatting" href="/wiki/Squatting">squatter</a>, who moves in with a group of radicals in London, and is drawn into their terrorist activities. Lessing began writing <i>The Good Terrorist</i> after the <a title="Provisional Irish Republican Army" href="/wiki/Provisional_Irish_Republican_Army">Irish Republican Army</a> (IRA) <a title="Harrods bombings" href="/wiki/Harrods_bombings">bombed</a> the <a title="Harrods" href="/wiki/Harrods">Harrods</a> department store in London in 1983. She had been a member of the <a title="Communist Party of Great Britain" href="/wiki/Communist_Party_of_Great_Britain">British Communist Party</a> in the early 1950s, but later grew disillusioned with it. Some reviewers labelled <i>The Good Terrorist</i> as a satire; Lessing called it humorous. Some focused on Alice's ambivalent nature, as highlighted by the novel's <a title="Oxymoron" href="/wiki/Oxymoron">oxymoronic</a> title, describing her as neither a good person nor a good revolutionary. Some were impressed by the book's insight and characterisation, while others complained about its style and the characters' lack of depth. The novel was shortlisted for the <a title="Man Booker Prize" class="mw-redirect" href="/wiki/Man_Booker_Prize">Booker Prize</a>, and won the Mondello Prize and the <a title="WH Smith Literary Award" href="/wiki/WH_Smith_Literary_Award">WH Smith Literary Award</a>. (<a title="The Good Terrorist" href="/wiki/The_Good_Terrorist"><b>Full article...</b></a>)</p>

<p><b><a title="February 8" href="/wiki/February_8">February 8</a></b></p>
<div id="mp-otd-img" style="float:right;margin-left:0.5em;">
<div style="background: transparent; border: none; padding: 0; max-width: 100px;" class="thumbinner mp-thumb"><a title="Mary, Queen of Scots" class="image" href="/wiki/File:Mary_Queen_of_Scots_portrait.jpg"><img width="100" height="130" data-file-height="500" data-file-width="386" srcset="//upload.wikimedia.org/wikipedia/commons/thumb/3/3f/Mary_Queen_of_Scots_portrait.jpg/150px-Mary_Queen_of_Scots_portrait.jpg 1.5x, //upload.wikimedia.org/wikipedia/commons/thumb/3/3f/Mary_Queen_of_Scots_portrait.jpg/200px-Mary_Queen_of_Scots_portrait.jpg 2x" src="//upload.wikimedia.org/wikipedia/commons/thumb/3/3f/Mary_Queen_of_Scots_portrait.jpg/100px-Mary_Queen_of_Scots_portrait.jpg" alt="Mary, Queen of Scots" /></a>
<div style="padding: 0.25em 0; word-wrap: break-word;" class="thumbcaption">Mary, Queen of Scots</div>
</div>
</div>
<ul>
<li><a title="1587" href="/wiki/1587">1587</a> – <b><a title="Mary, Queen of Scots" href="/wiki/Mary,_Queen_of_Scots">Mary, Queen of Scots</a></b> <i>(pictured)</i>, was executed at <a title="Fotheringhay Castle" href="/wiki/Fotheringhay_Castle">Fotheringhay Castle</a> for her involvement in the <a title="Babington Plot" href="/wiki/Babington_Plot">Babington Plot</a> to murder her cousin, <a title="Elizabeth I of England" href="/wiki/Elizabeth_I_of_England">Elizabeth I of England</a>.</li>
<li><a title="1855" href="/wiki/1855">1855</a> – A series of mysterious hoof-like marks known as the <b><a title="Devil's Footprints" href="/wiki/Devil%27s_Footprints">Devil's Footprints</a></b> appeared in the snow in <a title="Devon" href="/wiki/Devon">Devon</a>, England, and continued throughout the countryside for between 40 and 100 miles (64 and 161 km).</li>
<li><a title="1910" href="/wiki/1910">1910</a> – Newspaper and magazine publisher <b><a title="William D. Boyce" href="/wiki/William_D._Boyce">William D. Boyce</a></b> established the <a title="Boy Scouts of America" href="/wiki/Boy_Scouts_of_America">Boy Scouts of America</a>, expanding the <a title="Scouting" href="/wiki/Scouting">Scout Movement</a> into the United States.</li>
<li><a title="1960" href="/wiki/1960">1960</a> – The official groundbreaking for the <b><a title="Hollywood Walk of Fame" href="/wiki/Hollywood_Walk_of_Fame">Walk of Fame</a></b> took place in <a title="Hollywood, Los Angeles, California" class="mw-redirect" href="/wiki/Hollywood,_Los_Angeles,_California">Hollywood, Los Angeles, California</a>.</li>
<li><a title="2010" href="/wiki/2010">2010</a> – A freak storm in the <a title="Hindu Kush" href="/wiki/Hindu_Kush">Hindu Kush</a> mountains of Afghanistan triggered a series of at least <b><a title="2010 Salang avalanches" href="/wiki/2010_Salang_avalanches">36 avalanches</a></b> that buried over 3.5 km (2.2 mi) of road, killed at least 172 people and trapped over 2,000 travellers.</li>
</ul>

<td style="padding:0 0.9em 0 0;"><a title="Oenothera biennis" class="image" href="/wiki/File:Oenothera_rubricaulis_2014_G1.jpg"><img width="350" height="340" data-file-height="3450" data-file-width="3550" srcset="//upload.wikimedia.org/wikipedia/commons/thumb/0/0d/Oenothera_rubricaulis_2014_G1.jpg/525px-Oenothera_rubricaulis_2014_G1.jpg 1.5x, //upload.wikimedia.org/wikipedia/commons/thumb/0/0d/Oenothera_rubricaulis_2014_G1.jpg/700px-Oenothera_rubricaulis_2014_G1.jpg 2x" src="//upload.wikimedia.org/wikipedia/commons/thumb/0/0d/Oenothera_rubricaulis_2014_G1.jpg/350px-Oenothera_rubricaulis_2014_G1.jpg" alt="Oenothera biennis" /></a></td>
<td style="padding:0 6px 0 0">
<p><i><b><a title="Oenothera biennis" href="/wiki/Oenothera_biennis">Oenothera biennis</a></b></i> is a plant native to eastern and central North America and widely <a title="Naturalisation (biology)" href="/wiki/Naturalisation_%28biology%29">naturalized</a> elsewhere in <a title="Temperate climate" href="/wiki/Temperate_climate">temperate</a> and <a title="Subtropics" href="/wiki/Subtropics">subtropical</a> regions. It has a life span of <a title="Biennial plant" href="/wiki/Biennial_plant">two years</a>, growing to 30–150 cm (12–59 in) tall. The flowers are yellow and only last until the following noon after blooming. They open visibly fast every evening producing an interesting spectacle, hence the <a title="Common name" href="/wiki/Common_name">common name</a> "evening primrose."</p>
